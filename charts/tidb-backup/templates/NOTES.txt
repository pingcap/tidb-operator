{{- if .Values.fullbackup.create }}
1. Watch fullbackup cronjob is created
   watch kubectl get cronjobs --namespace {{ .Release.Namespace }} -l app.kubernetes.io/instance={{ .Release.Name }}
2. Wait until fullbackup pods becomes completed
   watch kubectl get jobs --namespace {{ .Release.Namespace }}  -l app.kubernetes.io/component=fullbackup
   watch kubectl get po --namespace {{ .Release.Namespace }}  -l app.kubernetes.io/component=fullbackup -o wide
{{- end -}}

   watch kubectl get jobs --namespace {{ .Release.Namespace }} -l app.kubernetes.io/instance={{ .Release.Name }} -o wide

   watch kubectl get po --namespace {{ .Release.Namespace }}  -l app.kubernetes.io/component=restore
